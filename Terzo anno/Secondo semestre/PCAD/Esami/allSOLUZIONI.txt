2022 Giugno 21 - ES 3
(Versione no var cond)
---------------------
Semaphore s(1);
int nReaders = 0;

read(){
   s.down();
   ++nReaders;
   s.up();
   read_val();
   s.down();
   --nReaders;
   s.up();
}

write(){
    while(true){
        s.down();
        if (nReaders == 0)
            break;
        s.up();
    }
    
   write_val();
   s.up();
}

Semaphore s(1);
int nReaders = 0;

read(){
   s.down();
   ++nReaders;
   s.up();
   read_val();
   s.down();
   --nReaders;
   if (nReaders == 0)
       signal(); // assocaito al semaforo???
   s.up();
}

write(){
    s.down();
    while(nReaders != 0)
        await(); // associato al semforo???
    
   write_val();
   s.up();
}
